    icon = GFX_idea_BRG_intensity_battle_situation_dynamic_modifier
    enable = {
        always = yes
    }
    remove_trigger = {
        OR = {
            set_temp_variable = { temp = 10 }
            BRG_war_situation_trigger = yes

#过载的运输状态
BRG_overloaded_transportation_status_dynamic_modifier = {
    icon = GFX_modifiers_BRA_neglected_state

    remove_trigger = {
        ROOT = {
            any_province_building_level = {
                province = {
                    all_provinces = yes
                }
                building = naval_base
                level > 7
            }
            infrastructure > 4
        }
    }

    army_speed_factor_for_controller = -0.25 #部队速度
    enemy_army_speed_factor = -0.1
    local_supplies = -0.15
}
