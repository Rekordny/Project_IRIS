#战局烈度
BRG_intensity_battle_situation_dynamic_modifier = {
    icon = GFX_idea_BRG_intensity_battle_situation_dynamic_modifier
    enable = {
        always = yes
    }
    remove_trigger = {
        OR = {
            set_temp_variable = { temp = 10 }
            BRG_war_situation_trigger = yes
            NOT = {
                has_country_flag = BRG_conflict_crisis_flag
                has_global_flag = BRG_2_a_flag
            }
        }
    }
    army_org_factor = BRG_intensity_battle_situation_army_org_factor #陆军组织度
    army_attack_factor = BRG_intensity_battle_situation_army_attack_factor #陆军攻击
    army_defence_factor = BRG_intensity_battle_situation_army_defence_factor #陆军防御
    experience_loss_factor = BRG_intensity_battle_situation_experience_loss_factor #有经验的士兵损失
}

#过载的运输状态
BRG_overloaded_transportation_status_dynamic_modifier = {
    icon = GFX_modifiers_BRA_neglected_state

    remove_trigger = {
        ROOT = {
            any_province_building_level = {
                province = {
                    all_provinces = yes
                }
                building = naval_base
                level > 7
            }
            infrastructure > 4
        }
    }

    army_speed_factor_for_controller = -0.25 #部队速度
    enemy_army_speed_factor = -0.1
    local_supplies = -0.15
}
