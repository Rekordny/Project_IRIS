#帝国四海承平
BLA_is_peaceful = {
    custom_trigger_tooltip = {
        tooltip = BLA_is_peaceful_desc
        NOT = {
            has_war_with = NAC
        }
        NOT = {
            country_exists = BLB
        }
        NOT = {
            has_country_flag = BLA_serfdom_question
        }
    }
}
