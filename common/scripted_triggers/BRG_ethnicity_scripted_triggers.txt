#判定战争态势
#接受临时变量
#temp 民族
BRG_war_situation_trigger = {
	custom_override_tooltip = {
		tooltip = {
			localization_key = BRG_WAR_SITUATION_TRIGGER_TOOLTIP
			INDENTATION = ""
			LEVEL = "[?temp]"
		}
		OR = {
			check_variable = { temp < global.BRG_war_escalation_level }
			check_variable = { temp = global.BRG_war_escalation_level }
		}
	}
}
