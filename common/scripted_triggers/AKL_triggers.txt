#皇国政府正常运行
AKL_everything_goes_well = {
    if = {
        limit = {
            has_country_leader = {
                character = AKL_Akari
            }
        }
        custom_trigger_tooltip = {
            tooltip = AKL_everything_goes_well_desc
            if = {
                limit = {
                    is_ai = no
                }
                check_variable = {AKL.AKL_authority > 29}
                check_variable = {AKL.AKL_popularity > 19}
                check_variable = {AKL.AKL_finance > 24}
                check_variable = {AKL.AKL_pressure < 80}
            }
        }
    }
}

AKL_has_authority_trigger = {
    custom_trigger_tooltip = {
        tooltip = AKL_has_authority_trigger_tt
        check_variable = {var = AKL_authority value = AKL_authority_temp compare = greater_than_or_equals}
    }
}
