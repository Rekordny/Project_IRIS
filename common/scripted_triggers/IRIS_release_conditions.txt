DIPLOMACY_RELEASE_NATION_ENABLE_TRIGGER = {
    if = {
		limit = {
			OR = {
				has_game_rule = {
					rule = allow_release_nations
					option = FREE
				}
				has_game_rule = {
					rule = allow_release_nations
					option = PEACE_ONLY
				}
			}
			OR = {
				FROM = { original_tag = CID }
			}
		}
		custom_trigger_tooltip = {
			tooltip = RULE_ALLOW_RELEASE_NATIONS_COSSACK_TOOLTIP
			NOT = {
				FROM = { tag = CID }
			}
		}
	}
}
