DIPLOMACY_RELEASE_NATION_ENABLE_TRIGGER = {
	if = {
		limit = {
			original_tag = KIV
		}
		always = yes
	}
	else = {
		custom_trigger_tooltip = {
			tooltip = RULE_ALLOW_RELEASE_NATIONS_COSSACK_TOOLTIP
			always = no
		}
	}
}
DIPLOMACY_REQUEST_EXP_FORCE_ENABLE_TRIGGER = {
	if = {
		limit = {
			OR = {
				#麦克阿瑟
				FROM = {
					has_country_leader = {
						character = USF_General_Douglas_MacArthur
						ruling_only = yes
					}
				}

				#塔露拉
				AND = {
					ROOT = {
						original_tag = OCF
						NOT = {
							has_country_leader = {
								character = OCF_Talulah_Arutorias
								ruling_only = yes
							}
						}
					}
					FROM = {
						OR = {
							has_country_leader = {
								character = OCF_Talulah_Arutorias
								ruling_only = yes
							}
							has_country_leader = {
								character = OCF_Alina
								ruling_only = yes
							}
						}
					}
				}
			}
		}
		custom_trigger_tooltip = {
			tooltip = IRIS_DIPLOMACY_REQUEST_EXP_FORCE_ENABLE_TRIGGER_TT
			always = no
		}
	}
}
