#判定主要民族
#接受临时变量
#temp_ethnicity 民族
#temp_name 目标 支持国家与省份作用域
IRIS_MAIN_ETHNIC_GROUP = {
	meta_trigger = {
		text = {
			custom_override_tooltip = {
				tooltip = {
					localization_key = IRIS_MAIN_ETHNIC_GROUP_TOOLTIP
					INDENTATION = ""
					NATIONAL = [ETHNICITY]
					NAME = [NAME]
				}
				OR = {
					check_variable = { [TAG].ethnicity = [TEMP_ETHNICITY] }
					check_variable = { [ID].ethnicity = [TEMP_ETHNICITY] }
				}
			}
		}
		TAG = "[?temp_name.GetTag]"
		ID = "[?temp_name.GetID]"
		NAME = "[?temp_name.GetName]"
		ETHNICITY = "[?temp_ethnicity.GetTokenLocalizedKey]"
		TEMP_ETHNICITY = "[?temp_ethnicity]"
	}
}


#是接纳的民族
#接受临时变量
#temp_ethnicity 民族
#temp_name 目标 支持国家与省份作用域
IRIS_ACCEPT_THIS_ETHNIC_GROUP = {
	meta_trigger = {
		text = {
			custom_override_tooltip = {
				tooltip = {
					localization_key = IRIS_ACCEPT_THIS_ETHNIC_GROUP_TOOLTIP
					INDENTATION = ""
					NATIONAL = [ETHNICITY]
					NAME = [NAME]
				}
				OR = {
					check_variable = { [TAG].ethnicity = [TEMP_ETHNICITY] }
					is_in_array = { [TAG].IRIS_race_array = [TEMP_ETHNICITY] }
				}
			}
		}
		TAG = "[?temp_name.GetTag]"
		ID = "[?temp_name.GetID]"
		NAME = "[?temp_name.GetName]"
		ETHNICITY = "[?temp_ethnicity.GetTokenLocalizedKey]"
		TEMP_ETHNICITY = "[?temp_ethnicity]"
	}
}
