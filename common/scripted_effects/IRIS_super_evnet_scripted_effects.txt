	set_country_flag = IRIS_super_evnet_center

	add_to_array = { IRIS_History_Array = se_display }

IRIS_fire_super_event = {
	hidden_effect = {
		if = {
			limit = {
				not = {
					has_country_flag = no_super_event_music
				}
			}
			IRIS_fire_super_event_music = yes
		}
	}
}
