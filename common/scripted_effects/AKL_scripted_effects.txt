        }
        swap_ideas = {

            add_idea = AKL_test_of_oomigami_3
        }

    else_if = {
        limit = {

        }
        swap_ideas = {
            remove_idea = AKL_test_of_oomigami_3
            add_idea = AKL_test_of_oomigami_4
        }
    }
    else_if = {
        limit = {
            has_idea = AKL_test_of_oomigami_4
        }
        if = {
            limit = {
                has_country_leader = {
                    character = AKL_Akari
                }
            }
            swap_ideas = {
                remove_idea = AKL_test_of_oomigami_4
                add_idea = AKL_oomigami_no_kuni_akari
            }
