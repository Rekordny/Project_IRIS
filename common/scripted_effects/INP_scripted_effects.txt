INP_athbheochan_initialize_effect = {
	set_variable = {INP_athbheochan_fire_var = 2}
	set_variable = {INP_athbheochan_fire_var_change = 1}
	set_variable = {INP_athbheochan_fire_display = 0}
	set_variable = {INP_athbheochan_stability = -0.05}
	set_variable = {INP_athbheochan_offence = 0}
	set_variable = {INP_athbheochan_army_speed = 0}
	set_variable = {INP_athbheochan_consumer_goods = 0}
	
	activate_mission = INP_athbheochan_update_timer
	add_dynamic_modifier = {modifier = INP_athbheochan_modifier}
	INP_athbheochan_refresh_effect = yes
}

INP_athbheochan_update_effect = {
	add_to_variable = {
		var = INP_athbheochan_fire_var
		value = INP_athbheochan_fire_var_change
	}
	INP_athbheochan_refresh_effect = yes
}
