MDS_liberty_effect_1 = {
	country_lock_all_division_template = no
	if = {
		limit = { has_idea = MDS_Philippe_Petain_idea }
		remove_ideas = MDS_Philippe_Petain_idea
	}
	if ={
		limit = {
			has_character = ELN_Philippe_Petain
		}
		ELN_Philippe_Petain = {
			set_nationality = ELN
		}
	}
	if = {
		limit = {
			check_variable = {
				MDS_whisper_lit_nm > 0
			}
			NOT = {
				has_country_flag = MDS_add_weapon_to_stockpile
			}
		}
		create_equipment_variant = {
			name = "Panzer II Ausf. M"
			type = light_tank_chassis_2
			allow_without_tech = yes
			parent_version = 0
			modules = {
				main_armament_slot = tank_small_cannon_2
				turret_type_slot = tank_light_three_man_tank_turret
				suspension_type_slot = tank_christie_suspension
				armor_type_slot = tank_riveted_armor
				engine_type_slot = tank_gasoline_engine
				special_type_slot_1 = tank_radio_1
				special_type_slot_2 = secondary_turret_hmg
				special_type_slot_3 = additional_machine_guns
				special_type_slot_4 = empty
			}
			upgrades = {
				tank_nsb_engine_upgrade = 0
				tank_nsb_armor_upgrade = 0
			}
			icon = "GFX_GER_improved_light_tank_medium"
			model = "GER_light_armor_entity"
		}
	}
}
