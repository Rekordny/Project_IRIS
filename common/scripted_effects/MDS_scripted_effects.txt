#更新元素共鸣
MDS_update_elemental_resonance = {
	#计算双元素满足条件
	#火元素
	if = {
		limit = {
			check_variable = {
				MDS_elemental_resonance_pyro_count > 1
			}
		}
		set_variable = {
			MDS_elemental_resonance_pyro = 0.25
		}
	}
	if = {
		limit = {
			check_variable = {
				MDS_elemental_resonance_pyro_count < 2
			}
		}
		set_variable = {
			MDS_elemental_resonance_pyro = 0
		}
	}
	#冰元素
	if = {
		limit = {
			check_variable = {
				MDS_elemental_resonance_cryo_count > 1
			}
		}
		set_variable = {
			MDS_elemental_resonance_cryo = 0.15
		}
	}
	if = {
		limit = {
			check_variable = {
				MDS_elemental_resonance_cryo_count < 2
			}
		}
		set_variable = {
			MDS_elemental_resonance_cryo = 0
		}
	}
	#岩元素
	if = {
		limit = {
			check_variable = {
				MDS_elemental_resonance_geo_count > 1
			}
		}
		set_variable = {
			MDS_elemental_resonance_geo_speed = 0.15
			MDS_elemental_resonance_geo_factor = 0.15
		}
	}
	if = {
		limit = {
			check_variable = {
				MDS_elemental_resonance_geo_count < 2
			}
		}
		set_variable = {
			MDS_elemental_resonance_geo_speed = 0
			MDS_elemental_resonance_geo_factor = 0
		}
	}
	#雷元素
	if = {
		limit = {
			check_variable = {
				MDS_elemental_resonance_electro_count > 1
			}
		}
		set_variable = {
			MDS_elemental_resonance_electro = 0.2
		}
	}
	if = {
		limit = {
			check_variable = {
				MDS_elemental_resonance_electro_count < 2
			}
		}
		set_variable = {
			MDS_elemental_resonance_electro = 0
		}
	}

	#计算是否有四种元素
	set_temp_variable = {
		MDS_elemental_resonance_count_tmp = 0
	}
	if = {
		limit = {
			check_variable = {
				MDS_elemental_resonance_pyro_count > 0
			}
		}
		add_to_temp_variable = {
			MDS_elemental_resonance_count_tmp = 1
		}
	}
	if = {
		limit = {
			check_variable = {
				MDS_elemental_resonance_cryo_count > 0
			}
		}
		add_to_temp_variable = {
			MDS_elemental_resonance_count_tmp = 1
		}
	}
	if = {
		limit = {
			check_variable = {
				MDS_elemental_resonance_geo_count > 0
			}
		}
		add_to_temp_variable = {
			MDS_elemental_resonance_count_tmp = 1
		}
	}
	if = {
		limit = {
			check_variable = {
				MDS_elemental_resonance_electro_count > 0
			}
		}
		add_to_temp_variable = {
			MDS_elemental_resonance_count_tmp = 1
		}
	}
	if = {
		limit = {
			check_variable = {
				MDS_elemental_resonance_count_tmp > 3
			}
		}
		set_variable = {
			MDS_elemental_resonance_count_if_four = 0.15
		}
	}
	if = {
		limit = {
			check_variable = {
				MDS_elemental_resonance_count_tmp < 4
			}
		}
		set_variable = {
			MDS_elemental_resonance_count_if_four = 0
		}
	}

	#添加/移除buff
	if = {
		limit = {
			check_variable = {
				MDS_elemental_resonance_electro_count < 2
			}
		}
		set_variable = {
			MDS_elemental_resonance_electro = 0
		}
	}
	if = {
		limit = {
			check_variable = {
				MDS_elemental_resonance_count < 4
			}
			has_dynamic_modifier = {
				modifier = MDS_elemental_resonance
				#scope = MDS
			}
		}
		remove_dynamic_modifier = {
			modifier = MDS_elemental_resonance
			#scope = MDS
		}
	}
	if = {
		limit = {
			check_variable = {
				MDS_elemental_resonance_count > 3
			}
		}
		add_dynamic_modifier = {
			modifier = MDS_elemental_resonance
			#scope = MDS
		}
	}
}

	}
