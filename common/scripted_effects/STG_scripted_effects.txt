        }

        }
        set_cosmetic_tag = STG_usr_subject_fascism
    }
    hidden_effect = {
		USR = { 
			add_ai_strategy = {
			    type = alliance
			    id = STG
			    value = 2000
			}
		}

			add_ai_strategy = {
			    type = alliance
			    id = USR
			    value = 2000
			}
		}
    }
