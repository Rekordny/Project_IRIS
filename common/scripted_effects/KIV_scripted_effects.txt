KIV_carve_up_melody = {
	ROOT = {
		transfer_navy = {
			target = KIV
		}
	}
	if = {
		limit = {
			ROOT = {
				is_faction_leader = yes
				all_allied_country = {
					is_major = no
				}
			}
		}
		every_country = {
			limit = {
				OR = {
					tag = ROOT
					is_ally_with = ROOT
				}
			}
			KIV = {
				annex_country = {
					target = PREV
					transfer_troops = no
				}
				release_autonomy = {
					target = PREV
					autonomy_state = autonomy_integrated_puppet
					freedom_level = 0.5
				}
			}
		}
	}
	else = {
		every_country = {
			limit = {
				OR = {
					tag = ROOT
					is_puppet_of = ROOT
				}
			}
			KIV = {
				annex_country = {
					target = PREV
					transfer_troops = no
				}
				release_autonomy = {
					target = PREV
					autonomy_state = autonomy_integrated_puppet
					freedom_level = 0.5
				}
			}
		}
	}
	every_country = {
		limit = {
			OR = {
				original_tag = KIV
				is_ally_with = KIV
			}
		}
		set_temp_variable = {
			temp = this
		}
		every_state = {
			limit = {
				is_owned_by = KIV
				NOT = {
					is_core_of = KIV
				}
				is_core_of = var:temp
			}
			var:temp = {
				set_state_owner = PREV
			}
		}
	}
	every_state = {
		limit = {
			is_owned_by = KIV
			NOT = {
				is_core_of = KIV
			}
		}
		add_to_temp_array = {
			temp_array = this
		}
	}
	every_possible_country = {
		limit = {
			OR = {
				exists = no
				is_puppet_of = KIV
			}
		}
		set_temp_variable = {
			temp = this
		}
		for_each_scope_loop = {
			array = temp_array
			if = {
				limit = {
					is_core_of = var:temp
				}
				KIV = {
					release_autonomy = {
						target = var:temp
						autonomy_state = autonomy_integrated_puppet
						freedom_level = 0.5
					}
				}
			}
		}
	}
	KIV_GSC_faction_array = yes
}
