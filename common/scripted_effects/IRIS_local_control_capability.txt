@PROGRESS_BAR_WIDTH = 256


#temp_levels_array #等级类别
#初始化地区控制力等级
IRIS_initialize_level_regional_control = {
    set_temp_variable = { temp_levels = temp_levels_array^num }
    add_to_temp_variable = { temp_levels = -1 }
    clear_array = scale_grid_array
    for_loop_effect = {
        start = 0
        end = temp_levels
        IRIS_add_segmentation_regional_control_level = yes
    }
    clear_array = regional_control_level_array
    for_loop_effect = {
        end = temp_levels_array^num
        value = i
        add_to_array = { regional_control_level_array = temp_levels_array^i }
    }
}
#添加地区控制力等级分段
IRIS_add_segmentation_regional_control_level = {
    add_to_array = { scale_grid_array = 1 }
    clear_array = scale_grid_array_x
    #x = (i + 1) * W / (n + 1)
    for_loop_effect = {
        start = 0
        end = scale_grid_array^num
        value = temp_i

        set_temp_variable = { i = temp_i }
        add_to_temp_variable = { i = 1 }
        set_temp_variable = { x = @PROGRESS_BAR_WIDTH }
        set_temp_variable = { n = scale_grid_array^num }
        add_to_temp_variable = { n = 1 }
        divide_temp_variable = { x = n }
        multiply_temp_variable = { x = i }
        add_to_array = { scale_grid_array_x = x }
    }

}
