#计算影响力
IRIS_impact_settlement = {
    #清空影响力
    clear_array = global.IRIS_effect_list
    set_variable = { global.IRIS_effect_list_total = 0 }
    #添加所有国家
    every_country = {
        limit = {
            NOT = {
                OR = {
                    has_country_flag = country_region_collapsed_flag
                    has_country_flag = has_Unmanned_area_flag
                    has_country_flag = has_Wasteland_flag
                    has_country_flag = has_Anarchy_flag
                    has_country_flag = has_Regional_riots_flag
                    has_country_flag = has_government_vacant_ocean_flag
                    TAG = WRQ
                    TAG = ANA
                    TAG = WST
                    TAG = ZTZ
                    TAG = RIR
                }
            }
            is_IRIS_TOUHOU_country = no
            is_IRIS_UB_country = no
            is_IRIS_LB_country = no
        }
        #工业与资源评估
        set_variable = { THIS.resource_score = 0 }
        
        #定义资源列表及其权重
        add_to_variable = { THIS.resource_score = THIS.resource_produced@oil }
        multiply_variable = { THIS.resource_score = 2.0 }
        
        set_variable = { THIS.temp_resource = THIS.resource_produced@aluminium }
        multiply_variable = { THIS.temp_resource = 1.2 }
        add_to_variable = { THIS.resource_score = THIS.temp_resource }
        
        set_variable = { THIS.temp_resource = THIS.resource_produced@rubber }
        multiply_variable = { THIS.temp_resource = 1.5 }
        add_to_variable = { THIS.resource_score = THIS.temp_resource }
        
        set_variable = { THIS.temp_resource = THIS.resource_produced@tungsten }
        multiply_variable = { THIS.temp_resource = 1.8 }
        add_to_variable = { THIS.resource_score = THIS.temp_resource }
        
        set_variable = { THIS.temp_resource = THIS.resource_produced@steel }
        multiply_variable = { THIS.temp_resource = 1.0 }
        add_to_variable = { THIS.resource_score = THIS.temp_resource }
        
        set_variable = { THIS.temp_resource = THIS.resource_produced@chromium }
        multiply_variable = { THIS.temp_resource = 1.8 }
        add_to_variable = { THIS.resource_score = THIS.temp_resource }
        
        set_variable = { THIS.temp_resource = THIS.resource_produced@raw_mana }
        multiply_variable = { THIS.temp_resource = 1.3 }
        add_to_variable = { THIS.resource_score = THIS.temp_resource }
        
        set_variable = { THIS.temp_resource = THIS.resource_produced@coal }
        multiply_variable = { THIS.temp_resource = 0.8 }
        add_to_variable = { THIS.resource_score = THIS.temp_resource }

        multiply_variable = { THIS.resource_score = 0.1 }

        
        set_variable = { THIS.national_influence = THIS.num_of_factories }
        add_to_variable = { THIS.national_influence = THIS.resource_score }


        #不通过省份来算了

        #人口与经济评估
        #总人口
        set_variable = { THIS.population_score = THIS.max_available_manpower_k }
        multiply_variable = { THIS.population_score = 0.05 }  # 进一步缩放
        
        # 合并人口经济分
        add_to_variable = { THIS.national_influence = THIS.population_score }
        



        set_variable = { THIS.national_influence_2 = THIS.num_subjects }
        multiply_variable = { THIS.national_influence_2 = 0.003 }
        add_to_variable = { THIS.national_influence_2 = 1 }
        multiply_variable = { THIS.national_influence = THIS.national_influence_2 }

        if = {
            limit = { is_faction_leader = yes }
            multiply_variable = { THIS.national_influence = 1.1 }
        }
        if = {
            limit = { is_puppet = yes }
            multiply_variable = { THIS.national_influence = 0.85 }
        }
        set_variable = { THIS.effect_tax_cap_2 = 1 }
        add_to_variable = { THIS.effect_tax_cap_2 = modifier@effect_tax_cap }
        multiply_variable = { THIS.national_influence = THIS.effect_tax_cap_2 }
        multiply_variable = { THIS.national_influence = 100 }
        add_to_array = { global.IRIS_effect_list = THIS }

        #记录总分
        add_to_variable = { global.IRIS_effect_list_total = THIS.national_influence }
    }
    #计算得分器
    get_sorted_scored_countries = {
        scorer = IRIS_effect_list_scorer
        array = global.IRIS_effect_list
    }
    #循环计算排名
    for_each_scope_loop = {
        array = global.IRIS_effect_list
        add_to_variable = { global.IRIS_effect_list_index = 1 }
        set_variable = { IRIS_effect_list_2 = global.IRIS_effect_list_index }
        
        #计算国家排名称号
        set_variable = { THIS.national_influence_proportion = THIS.national_influence }
        divide_variable = { THIS.national_influence_proportion = global.IRIS_effect_list_total }
    }
    clear_variable = global.IRIS_effect_list_index
    IRIS_refresh_troop_limit = yes

}

add_integrable_core = {
    custom_effect_tooltip = integratable_regions

    add_to_array = {
        FROM.add_integrable_core = PREV
    }

    FROM = {
        set_country_flag = integratable_regions_yes
    }

    FROM.FROM = {
        custom_effect_tooltip = add_integrable_core_tooltip
        every_controlled_state = {
            set_occupation_law = integratable_regions
        }
    }
}

add_integrable_core_80 = {
    FROM = {
        set_temp_variable = { integrator_target = THIS }
    }
    if = {
        limit = { has_variable = integratable_regions_tag }
        if = {
            limit = { NOT = { check_variable = { can_integratable_regions@var:integratable_regions_tag = 80 } } }
            set_variable = { can_integratable_regions@var:integratable_regions_tag = 80 }
            if = {
                limit = {
                    is_controlled_by = var:integratable_regions_tag
                    compliance > 80
                }
                add_core_of = var:integratable_regions_tag
            }
        }
    }
    else = {
        if = {
            limit = {
                is_controlled_by = PREV
                compliance > 80
            }
            add_core_of = PREV
        }
    }
    FROM.FROM = {
        custom_effect_tooltip = add_integrable_core_tooltip
        every_controlled_state = {
            set_occupation_law = integratable_regions
        }
    }
}
