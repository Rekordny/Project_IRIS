scripted_gui = {
	map_mode_window = {
		context_type = player_context

		window_name = "map_mode_window"
		parent_window_token = top_bar

		mapmode = dimai_map_mode

		visible = {
			is_ai = no
		}

		effects = {
			button_click = {
				if = {
					limit = {
						has_country_flag = disable_map_mode_mapicons
					}
					clr_country_flag = disable_map_mode_mapicons
				}
				else = {
					set_country_flag = disable_map_mode_mapicons
				}
			}
		}

		triggers = {
			button_visible = {
				if = {
					limit = {
						has_country_flag = disable_map_mode_mapicons
					}
					set_temp_variable = { button_frame = 1 }
				}
				else = {
					set_temp_variable = { button_frame = 2 }
				}
			}
		}

		properties = {
			button = {
				frame = button_frame
			}
		}
	}
}
