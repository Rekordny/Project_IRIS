	#一次转移焦点的战争行动
	ANZ_war_to_shift_focus = {
		visible = {
			has_completed_focus = ANZ_trial_of_foreigners
		}
		available = {
			TAU = {
				all_core_state = {
					is_controlled_by = ANZ
				}
			}
		}
		fire_only_once = yes
		complete_effect = {
			add_power_balance_value = {
				id = ANZ_industrialization_against_manalisation
				value = -0.5
			}
			TAU = {
				every_core_state = {
					add_compliance = 20
				}
			}
		}
	}
	#彻底清除地脉
	ANZ_remove_dimai_completely = {
		visible = {
			#has_completed_focus = ANZ_treaty_of_saint_michaeldonia
		}
		available = {
			has_idea = ANZ_ancient_mana_tradition
			ANZ = {
				all_core_state = {
					is_controlled_by = ANZ
					dimai < 1
				}
			}
		}
		fire_only_once = yes
		complete_effect = {
			remove_ideas = ANZ_ancient_mana_tradition
		}
	}
	#摧毁地脉
	ANZ_destroy_dimai = {
		cost = 15
		modifier = {
			civilian_factory_use = 1
		}
		visible = {
			#has_completed_focus = ANZ_treaty_of_saint_michaeldonia
		}
		# fire_only_once = yes
		days_remove = 30
		complete_effect = {
			add_power_balance_value = {
				id = ANZ_industrialization_against_manalisation
				value = 0.02
			}
			if = {
				limit = { has_country_flag = ANZ_LOTHIAN }
				custom_effect_tooltip = ANZ_tooltip5
				add_to_variable = {
					anz_civil_our_power = -0.3
				}
			}
			else = {
				custom_effect_tooltip = ANZ_tooltip5
				add_to_variable = {
					anz_civil_enemy_power = 0.3
				}
			}
		}
		remove_effect = {
			random_core_state = {
				limit = { is_controlled_by = ANZ dimai > 0 }
				remove_building = {
					type = dimai
					level = 5
				}
			}
		}
	}
	#集中销毁玛娜制品
	ANZ_destroy_mana_artifacts = {
		cost = 30
		visible = {
			has_completed_focus = ANZ_destroy_mana_artifacts
		}
		# fire_only_once = yes
		days_remove = 100
		complete_effect = {
			add_power_balance_value = {
				id = ANZ_industrialization_against_manalisation
				value = 0.05
			}

				limit = { has_country_flag = ANZ_LOTHIAN }
				custom_effect_tooltip = ANZ_tooltip5
				add_to_variable = {
					anz_civil_our_power = -1
				}
			}
			else = {
				custom_effect_tooltip = ANZ_tooltip5
				add_to_variable = {
					anz_civil_enemy_power = 1
				}
			}
		}
		remove_effect = {
			random_core_state = {
				limit = { is_controlled_by = ANZ dimai > 0 }
				remove_building = {
					type = dimai
					level = 5
				}
			}
		}
	}
	#允许礼拜税
	ANZ_allow_liturgical_tax = {
		visible = {
			has_completed_focus = ANZ_treaty_of_saint_michaeldonia
		}
		fire_only_once = yes
		complete_effect = {
			add_power_balance_value = {
				id = ANZ_industrialization_against_manalisation
				value = -0.15
			}
			add_ideas = ANZ_extra_fiscal_revenue_of_holy_see
			if = {
				limit = { has_country_flag = ANZ_LOTHIAN }
				custom_effect_tooltip = ANZ_tooltip6
				add_to_variable = {
					anz_civil_our_power = 1
				}
				
			}
			else = {
				custom_effect_tooltip = ANZ_tooltip5
				add_to_variable = {
					anz_civil_enemy_power = 1
				}
			}
		}
	}
	#规整地方牧区
	ANZ_file_local_parish = {
		visible = {
			has_completed_focus = ANZ_treaty_of_saint_michaeldonia
		}
		cost = 25
		fire_only_once = no
		days_remove = 100
		complete_effect = {
			add_stability = 0.01
			add_power_balance_value = {
				id = ANZ_industrialization_against_manalisation
				value = -0.03
			}
			if = {
				limit = { has_country_flag = ANZ_LOTHIAN }
				custom_effect_tooltip = ANZ_tooltip6
				add_to_variable = {
					anz_civil_our_power = 0.5
				}
				
			}
			else = {
				custom_effect_tooltip = ANZ_tooltip5
				add_to_variable = {
					anz_civil_enemy_power = 0.5
				}
			}
		}
	}
	
	#减缓工业化
	ANZ_slow_the_industrial_trend = {
		cost = 25
		remove_effect = {
			random_list = {
				50 = {

				}
				25 = {
					country_event = { id = anz.3 }
				}
				25 = {
					if = {
						#如果已经累计10个bop，事实上你已经快完蛋了
						limit = { check_variable = {
							anz_industrialization_extra_var < 1
						} }
						country_event = { id = anz.4 }
					}
					else = {
						add_stability = -0.01
					}
				}
			}
		}
		days_remove = 100  
		complete_effect = {
			add_power_balance_value = {
				id = ANZ_industrialization_against_manalisation
				value = -0.05
			}
			
		}
		ai_will_do = {
			base = 0
			modifier = {
				has_country_flag = ANZ_ANZIO
				power_balance_value = {
					id = ANZ_industrialization_against_manalisation
					value > 0.5
				}
				NOT = { has_completed_focus = ANZ_ensure_the_loyalty_of_the_officers }
				add = 5
			}
			modifier = {
				has_country_flag = ANZ_LOTHIAN
				power_balance_value = {
					id = ANZ_industrialization_against_manalisation
					value > 0.5
				}
				NOT = { has_completed_focus = ANZ_icorporation_of_diocese_of_aldrich }
				add = 5
			}
		}
	}
	#移除工厂
	ANZ_remove_factory = {
		cost = 0
		remove_effect = {
			random_controlled_state = {
				remove_building = {
					type = industrial_complex
					level = 1
				}
			}
		}
		days_remove = 100  
		complete_effect = {
			add_political_power = 50
			add_power_balance_value = {
				id = ANZ_industrialization_against_manalisation
				value = -0.05
			}
			
		}
		ai_will_do = {
			base = 0
			modifier = {
				has_country_flag = ANZ_ANZIO
				power_balance_value = {
					id = ANZ_industrialization_against_manalisation
					value > 0.7
				}
				NOT = { has_completed_focus = ANZ_ensure_the_loyalty_of_the_officers }
				add = 1
			}
			modifier = {
				has_country_flag = ANZ_LOTHIAN
				power_balance_value = {
					id = ANZ_industrialization_against_manalisation
					value > 0.7
				}
				NOT = { has_completed_focus = ANZ_icorporation_of_diocese_of_aldrich }
				add = 1
			}
		}
	}
}

ANZ_secret_senshado_cat = {
	#藏匿军用装备
	ANZ_gather_equipment = {
		fire_only_once = no
		days_remove = 100
		custom_cost_trigger = {
			has_army_experience > 30
		}
		visible = {
			has_civil_war = no
			has_completed_focus = ANZ_ignite_the_black_flame
		}
		custom_cost_text = ANZ_tooltip15
		complete_effect = {
			hidden_effect = {
				army_experience = -30
			}
		}
		remove_effect = {
			add_equipment_to_stockpile = {
				type = infantry_equipment
				amount = -2000
			}
			add_equipment_to_stockpile = {
				type = support_equipment
				amount = -100
			}
			add_to_variable = {
				anz_gather_equipment_var = 1
			}
			custom_effect_tooltip = ANZ_tooltip16
		}
		ai_will_do = {
			factor = 10
		}
	}
	#将之前藏匿的军用装备投入使用
	ANZ_reuse_equipment = {
		visible = {
			has_civil_war = yes
			check_variable = {
				anz_gather_equipment_var > 0
			}
		}
		complete_effect = {
			custom_effect_tooltip = ANZ_tooltip17
			while_loop_effect = {
				limit = { check_variable = {
					ANZ.anz_gather_equipment_var > 0
				} }
				add_to_variable = {
					ANZ.anz_gather_equipment_var = -1
				}
				add_equipment_to_stockpile = {
					type = infantry_equipment
					amount = -2500
					producer = ANZ
				}
				add_equipment_to_stockpile = {
					type = support_equipment
					amount = -100
					producer = ANZ
				}
			}
		}
	}
	
}
