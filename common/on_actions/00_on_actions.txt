on_actions = {
# ROOT is capitulated country, FROM is winner
	on_capitulation = {
		effect = {
			if = {
				limit = {
					has_civil_war = no
					NOT = {
						AND = {
							FROM = { original_tag = PRM }
                            ROOT = { original_tag = REC }
						}
					}
					NOT = {
						AND = {
							FROM = { original_tag = BLA }
                            ROOT = {
								OR = {
									original_tag = NAC
									original_tag = ORE
									original_tag = HEL
									original_tag = OPA
								}
							}
						}
					}
					NOT = {
						AND = {
							ROOT = { original_tag = BLA }
                            FROM = {
								OR = {
									original_tag = NAC
									original_tag = ORE
									original_tag = HEL
									original_tag = OPA
								}
							}
						}
					}
					NOT = {
						AND = {
							FROM = { original_tag = OPA }
                            FROM = {
								OR = {
									original_tag = NAC
									original_tag = ORE
									original_tag = HEL
									original_tag = BLA
								}
							}
						}
					}
					NOT = {
						AND = {
							has_global_flag = WAR_FOR_ANGLIA
							ROOT = { original_tag = STG }
                            FROM = {
								OR = {
									original_tag = UNG
									original_tag = FNG
									original_tag = MSA
								}
							}
						}
					}
					NOT = {
						AND = {
							has_global_flag = WAR_FOR_ANGLIA
							FROM = { original_tag = STG }
                            ROOT = {
								OR = {
									original_tag = UNG
									original_tag = FNG
									original_tag = MSA
								}
							}
						}
					}
				}
				news_event = { id = country_capitulated.0 days = 0 }
			}
            if = {
				limit = {
					FROM = { original_tag = PRM }
					ROOT = { original_tag = REC }
				}
				news_event = { id = country_capitulated.1 days = 0 }
			}
		}
	}
}
