on_actions = {
	on_startup = {
		effect = {
			if = {
				limit = {
					tag = USR
				}
				#共和议会进程初始化
				USR = {
					set_variable = { usr_congress = 1 }
				}
			}
		}
	}
	on_offer_join_faction = {
		effect = {
			USR_refresh_SSD_alliance_members = yes
		}
	}
	on_monthly_USR = {
		effect = {
			USR_refresh_faction_effect = yes
		}
	}
	on_war = {
		effect = {
			if = {
				limit = {
					tag = USR
					has_war_with = BGT
				}
				country_event = { id = usr.38 }
			}
		}
	}
}
