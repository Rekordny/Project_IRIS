	on_declare_war = {
		effect = {
			if = {
				limit = {
					tag = USR
					FROM = { tag = MDO }
					MDS = { 
						is_puppet_of = MDO 
						has_completed_focus = MDS_focus_dip_12
					} 
				}
				MDO = {
					set_autonomy = {
						target = MDS
						autonomous_state = autonomy_free
						end_wars = yes
					}
					remove_from_faction = MDS
				}
				if = {
					limit = {
						is_in_faction = yes
					}
					faction_leader = {
						add_to_faction = MDS
					}
				}
				MDS = {
					declare_war_on = {
						target = MDO
						type = annex_everything
					}
				}
				country_event = MDS.11
				MDS = {
					country_event = MDS.11
				}
				MDO = {
					country_event = MDS.11
				}
			}
			if = {
				limit = {
					original_tag = MDS
					FROM = {
						original_tag = MDO
					}
				}
				MDS = {
					set_country_flag = MDS_free_from_ELN
