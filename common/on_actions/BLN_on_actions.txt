on_actions = {
    on_capitulation = {
        effect = {
            if = {
                limit = {
                    ROOT = {
                        tag = BLN
                    }
                    FROM = {
                        OR = {
                            tag = SCA
                            tag = DKG
                        }
                    }
                }
                set_global_flag = show_peace_popup_alert
				set_global_flag = war_won
				FROM = {
					save_global_event_target_as = winning_country
				}
				ROOT = {
					save_global_event_target_as = losing_country
				}
				IRIS_peace_popup_determine_allies = yes
				add_to_variable = { global.peace_popup_dirty = 1 }
                SCA = {
                    add_ideas = SCA_shadow_of_DKG
                    annex_country = {
                        target = ROOT
                    }
                    annex_country = {
                        target = DKG
                    }
                    set_cosmetic_tag = SCA_reclaimed_BLN
                }
                every_state = {
                    limit = {
                        is_core_of = ROOT
                    }
                    add_core_of = FROM
                }

                set_global_flag = BLN_CIV_WAR_ENDED
            }
            if = {
                limit = {
                    ROOT = {
                        OR = {
                            tag = SCA
                            tag = DKG
                        }
                    }
                    FROM = {
                        tag = BLN
                    }
                }
                set_global_flag = show_peace_popup_alert
				set_global_flag = war_won
				FROM = {
					save_global_event_target_as = winning_country
				}
				ROOT = {
					save_global_event_target_as = losing_country
				}
				IRIS_peace_popup_determine_allies = yes
				add_to_variable = { global.peace_popup_dirty = 1 }
                FROM = {
                    annex_country = {
                        target = ROOT
                    }
                }
                every_state = {
                    limit = {
                        is_core_of = ROOT
                    }
                    add_core_of = FROM
                    remove_core_of = ROOT
                }

                set_global_flag = BLN_CIV_WAR_ENDED
            }
            if = {
                limit = {
                    ROOT = {
                        OR = {
                            tag = BLN
                            tag = DKG
                        }
                    }
                    FROM = {
                        OR = {
                            tag = YMI
                            is_in_faction_with = YMI
                        }
                    }
                    has_global_flag = YMI_INTERVENE_BLN_FLAG
                }
                set_global_flag = show_peace_popup_alert
				set_global_flag = war_won
				FROM = {
					save_global_event_target_as = winning_country
				}
				ROOT = {
					save_global_event_target_as = losing_country
				}
				IRIS_peace_popup_determine_allies = yes
				add_to_variable = { global.peace_popup_dirty = 1 }
                FROM = {
                    annex_country = {
                        target = ROOT
                    }
                }
            }
        }
    }
}
