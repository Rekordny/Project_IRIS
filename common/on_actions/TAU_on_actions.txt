on_actions = {
	on_startup = {
		effect = {
			TAU = {
				#女神像数量初始化
				set_variable = { tau_nsx = 0 }
				#发展度初始化
				set_variable = { tau_dev = 0 }
				#发展度国土基数初始化
				set_variable = { tau_dev_base_factor = 0 }
				#发展度奖励基数初始化
				set_variable = { tau_dev_bonus_factor = 0 }
				#更新国土基数并更新发展度
				TAU_update_dev_base_factor = yes
				#更新女神像获取
				TAU_update_nsx = yes
			}
		}
	}
	
	on_weekly_TAU = { 
		effect = {	
			TAU = {
				#更新国土基数并更新发展度
				TAU_update_dev_base_factor = yes
				#更新女神像获取
				TAU_update_nsx = yes
			}
		}
	}
}
