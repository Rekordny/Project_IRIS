on_actions = {
    on_startup = {
        effect = {
            every_country = {
                #动态图书馆加成修正
                set_variable = {
                    PRM_book = 0
                }
                #书本数量
                set_variable = {
                    PRM_book_stability_factor = 0
                }
                #稳定度
                set_variable = {
                    PRM_book_war_support_factor = 0
                }
                #战争支持度
                set_variable = {
                    PRM_book_land_reinforce_rate = 0
                }
                #增援率
                set_variable = {
                    PRM_book_army_attack_factor = 0
                }
                #攻击
                set_variable = {
                    PRM_book_army_defence_factor = 0
                }
                #防御
                set_variable = {
                    PRM_book_army_morale_factor = 0
                }
                #恢复
                set_variable = {
                    PRM_book_army_org_factor = 0
                }
                #组织度
                set_variable = {
                    PRM_book_attrition = 0
                }
                #堑壕速度
                set_variable = {
                    PRM_book_breakthrough_factor = 0
                }
                #最大堑壕
                set_variable = {
                    PRM_book_army_speed_factor = 0
                }
                #速度
                set_variable = {
                    PRM_book_max_planning_factor = 0
                }
                #最大规划
                set_variable = {
                    PRM_book_terrain_penalty_reduction = 0
                }
                #地形惩罚减少
                set_variable = {
                    PRM_book_political_power_factor = 0
                }
                #每日PP获取
                PRM = {
                    add_dynamic_modifier = {
                        modifier = PRM_morale_modifier
                    }
                    #动态图书馆加成修正
                }
            }
        }
    }
    on_capitulation_immediate = {
        effect = {
            #你永远打不死图书馆
            if = {
                limit = {
                    ROOT = {
                        original_tag = PRM
                    }
                }
                ROOT = {
                    white_peace = FROM
                }
                country_event = {
                    id = PRM_reception_country.4
                }
            }

            #你永远打不死图书馆
            if = {
                limit = {
                    ROOT = {
                        original_tag = REC
                    }
                }
                if = {
                    limit = {
                        FROM = {
                            original_tag = PRM
                        }
                    }
                    FROM = {
                        white_peace = {
                            tag = ROOT
                            message = PRM_REC_WAR_OVER_TXT
                        }
                        annex_country = {
                            target = ROOT transfer_troops = no
                        }
                    }
                }
                every_state = {
                    limit = {
                        is_core_of = ROOT
                    }
                    ROOT = {
                        set_state_controller = PREV
                        transfer_state = PREV
                    }
                }
            }
        }
    }
}
