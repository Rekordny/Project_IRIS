    #和平会议傀儡
    on_puppet = {
        effect = {
            ROOT = {
                if = {
                    limit = {
                        NOT = { has_variable = { root.ethnicity = this.ethnicity } }
                    }
                    log = "民族检测：[ROOT.GetName]和平会议傀儡"
                    random_country = {
                        limit = {
                            original_tag = root
                            has_variable = ethnicity
                        }
                        log = "民族检测：[ROOT.GetName]和平会议傀儡复制[this.GetName]民族信息"
                        set_variable = { root.ethnicity = this.ethnicity }
                        clear_array = root.IRIS_race_array
                        for_each_loop = {
                            array = this.IRIS_race_array
                            value = v
                            add_to_array = { root.IRIS_race_array = v }
                        }
                    }
                }
                IRIS_add_national_ethnicity = yes
                IRIS_country_list_testing_region = yes
                default_policy_one = yes
            }
        }
    }
