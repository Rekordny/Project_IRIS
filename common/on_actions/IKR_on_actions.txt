on_actions = {
	on_startup = {
		effect = {
			IKR = {
				if = {
					limit = {
						is_ai = no
					}
					set_variable = {IKR_weapon = 50}
					set_variable = {IKR_money = 50}
					set_variable = {IKR_food = 50}
				}
			}
		}
	}
	on_weekly_IKR = {
		effect = {
			IKR = {
				if = {
					limit = {
						has_idea = IKR_revolution
						is_ai = no
					}
					IKR_trigger_random_regional_event = yes
					IKR_check_w_f_m = yes
				}
			}
		}
	}
	on_monthly_IKR = {
		effect = {
			IKR = {
				if = {
					limit = {
						has_idea = IKR_new_contract
					}
					MRS = {
						add_equipment_to_stockpile = {
							type = infantry_equipment_2
							amount = 2000
						}
						send_equipment = {
							equipment = infantry_equipment_2
							amount = 1000
							target = IKR
						}
					}
					add_equipment_to_stockpile = {
						type = support_equipment_1
						amount = 100
					}
				}
				if = {
					limit = {
						has_idea = IKR_old_contract
					}
					add_equipment_to_stockpile = {
						type = infantry_equipment_1
						amount = 2000
					}
					add_equipment_to_stockpile = {
						type = support_equipment_1
						amount = 100
					}
				}
			}
		}
	}
    on_capitulation = {
		effect = {
			if = {
				limit = {
					ROOT = {
						OR = {
							tag = KAM
							tag = BAT
						}
						has_war_with = IKR
					}
				}
				if = {
					limit = {
						FROM = {
							tag = IKR
						}
					}
					log = "[GetDateText]: FROM: [From.GetName] ROOT: [Root.GetName] on_capitulation"
					set_global_flag = show_peace_popup_alert
					set_global_flag = war_won
					IKR = {
						save_global_event_target_as = winning_country
					}
					ROOT = {
						save_global_event_target_as = losing_country
					}
					IRIS_peace_popup_determine_allies = yes
					add_to_variable = { global.peace_popup_dirty = 1 }
					FROM = {
						annex_country = {
							target = ROOT
							transfer_troops = no
                        }
						if = {
							limit = {
								has_war_with = MZL
							}
							white_peace = MZL
						}
					}
				}
			}
			if = {
				limit = {
					ROOT = {
						tag = MZL
						has_war_with = IKR
					}
				}
				if = {
					limit = {
						FROM = {
							tag = IKR
						}
					}
					log = "[GetDateText]: FROM: [From.GetName] ROOT: [Root.GetName] on_capitulation"
					set_global_flag = show_peace_popup_alert
					set_global_flag = war_won
					IKR = {
						save_global_event_target_as = winning_country
					}
					MZL = {
						save_global_event_target_as = losing_country
					}
					IRIS_peace_popup_determine_allies = yes
					add_to_variable = { global.peace_popup_dirty = 1 }
					FROM = {
						annex_country = {
							target = ROOT
							transfer_troops = no
                        }
					}
					every_possible_country = {
						limit = {
							has_character = MZL_Bhagyate
						}
						MZL_Bhagyate = {
							set_nationality = IKR
						}
					}
					every_possible_country = {
						limit = {
							has_character = BAT_Aishe
						}
						BAT_Aishe = {
							set_nationality = IKR
						}
					}
					every_state = {
						limit = {
							is_core_of = MZL
						}
						remove_core_of = MZL
					}
					news_event = {id = IKR_news.1}
				}
			}
			if = {
				limit = {
					FROM = {
						OR = {
							tag = MZL
							is_in_faction_with = MZL
						}
					}
					ROOT = {
						OR = {
							tag = IKR
							tag = KAM
							tag = KLC
						}
					}
				}
				log = "[GetDateText]: FROM: [From.GetName] ROOT: [Root.GetName] on_capitulation"
				set_global_flag = show_peace_popup_alert
				set_global_flag = war_won
				FROM = {
					save_global_event_target_as = winning_country
				}
				ROOT = {
					save_global_event_target_as = losing_country
				}
				IRIS_peace_popup_determine_allies = yes
				add_to_variable = { global.peace_popup_dirty = 1 }
				MZL = {
					annex_country = {
						target = ROOT
						transfer_troops = no
					}
				}
			}
        }
    }
}
