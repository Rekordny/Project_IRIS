on_actions = {
	on_startup = {
		effect = {
			ELN = {
				country_event = {
					id = elino.1
					days = 10
				}
				if = {
					limit = {
						not = {
							has_idea = ELN_Master_of_eu
						}
					}
					add_ideas = ELN_Master_of_eu
				}
				ELN_Load_Workers_Council = yes
				set_variable = { ELN_Great_Society_Freedom_Var = 0 }
				set_variable = { ELN_Great_Society_Equal_Var = 0 }
				set_variable = { ELN_Great_Society_Fraternity_Var = 0 }
				set_variable = { ELN_cor_with_right_window_var = 1 }
			}
			127 = { set_state_flag = has_city_story }
		}
	}
	on_monthly_ELN = {
		effect = {
			if = {
				limit = {
					has_global_flag = Entente_for_Democracies_faction
				}
				Entente_for_Democracies_array_effect = yes
			}
			ELN = {
				if = {
					limit = {
						has_dynamic_modifier = { modifier = ELN_Great_Society_modifier }
					}
					add_to_variable = { ELN_Great_Society_Freedom_Var = ELN_Great_Society_Freedom_Var_monthly }
					add_to_variable = { ELN_Great_Society_Equal_Var = ELN_Great_Society_Equal_Var_monthly }
					add_to_variable = { ELN_Great_Society_Fraternity_Var = ELN_Great_Society_Fraternity_Var_monthly }
					ELN_flushed_Great_Society = yes
				}
			}
		}
	}
	on_daily_ELN = {
		effect = {
			ELN = {
				if = {
					limit = {
						has_completed_focus = ELN_ECP_The_first_five_year_plan
						not = { has_country_flag = has_completeed_First_Years_Plan }
					}
					ELN_flushed_First_Years_Plan = yes
				}
			}
		}
	}
	on_weekly_ELN = {
		effect = {
		}
	}
	on_capitulation_immediate = {
		effect = {
			if = {
				limit = {
					ROOT = {
						tag = EBR
					}
					FROM = {
						tag = ELN
					}
				}
				EBR = {
					faction_leader = {
						dismantle_faction = yes
					}
					transfer_navy = {
						target = ELN
					}
				}
				ELN = {
					white_peace = {
						tag = EBR
						message = ELN_WIN_EBR_TT
					}
					annex_country = { target = EBR transfer_troops = yes }
					country_event = {
						id = elino.1003
					}
					country_event = {
						id = elino.27
					}
				}
				every_country = {
					limit = {
						is_core_of = EBR
					}
					remove_core_of = EBR
				}
			}
		}
	}
}
