#瘟疫传播
country_event = {
    id = USF_pandemic.1
    title = USF_pandemic.1.t
	desc = USF_pandemic.1.desc
	#picture = GFX_event_USF_1
    immediate = {
        hidden_effect = {
            USF_pandemic_spread = yes
        }
    }
	trigger = {
        tag = USF
        any_core_state = {
            has_dynamic_modifier = {modifier = USF_pandemic scope = USF}
            any_neighbor_state = {
                is_core_of = USF
                NOT = {
                    OR = {
                        is_island_state = yes
                        has_state_category = small_island
                        has_state_category = tiny_island
                    }
                }
                NOT = {has_dynamic_modifier = {modifier = USF_pandemic scope = USF}}
            }
        }
    }
    mean_time_to_happen = {
        days = var:USF_pandemic_spread?25
    }

	option = {
		name = USF_pandemic.1.a

        event_target:USF_pandemic_target = {
            add_dynamic_modifier = {modifier = USF_pandemic scope = USF}
            if = {
                limit = {
                    NOT = {
                        is_in_array = {
                            array = USF.pandemic_states
                            value = THIS.id
                        }
                    }
                }
                add_to_array = {
                    array = USF.pandemic_states
                    value = THIS.id
                }
            }
        }
        add_to_variable = {USF.USF_pandemic_count = 1}

		ai_chance = {
			base = 10
		}
	}
}
