	id = STG_news.5
	title = STG_news.5.t
	desc = STG_news.5.desc
	picture = GFX_Current_War
	major = yes
	immediate = {
		set_global_flag = WAR_FOR_ANGLIA
	}
	trigger = {
		if = {
			limit = {
				country_exists = UNG
			}
			STG = {has_war_with = UNG}
		}
		else_if = {
			limit = {
				country_exists = FNG
			}
			STG = {has_war_with = FNG}
		}
		else = {
			STG = {has_war_with = MSA}
		}
		NOT = {
			STG = {is_puppet = yes}
			UNG = {is_puppet = yes}
			MSA = {is_puppet = yes}
			FNG = {is_puppet = yes}
		}
		NOT = {
			has_global_flag = WAR_FOR_ANGLIA
		}
		NOT = {
			has_global_flag = STG_operation_long_bow
		}
	}
	mean_time_to_happen = {
		days = 5
	}

	option = {
		name = STG_news.5.a
		hidden_effect = {
			if = {
				limit = {
					country_exists = UNG
					NOT = {
						country_exists = MSA
					}
					NOT = {
						country_exists = FNG
					}
					STG = {has_war_with = UNG}
				}
				if = {
					limit = {
						tag = STG
					}
					country_event = {id = STG_war.4}
				}
				if = {
					limit = {
						tag = UNG
					}
					country_event = {id = STG_war.4}
				}
			}
			if = {
				limit = {
					country_exists = MSA
					NOT = {
						country_exists = UNG
					}
					NOT = {
						country_exists = FNG
					}
					STG = {has_war_with = MSA}
				}
				if = {
					limit = {
						tag = STG
					}
					country_event = {id = STG_war.4}
				}
				if = {
					limit = {
						tag = MSA
					}
					country_event = {id = STG_war.4}
				}
			}
			if = {
				limit = {
					country_exists = FNG
					NOT = {
						country_exists = UNG
					}
					NOT = {
						country_exists = MSA
					}
					STG = {has_war_with = FNG}
				}
				if = {
					limit = {
						tag = STG
					}
					country_event = {id = STG_war.4}
				}
				if = {
					limit = {
						tag = FNG
					}
					country_event = {id = STG_war.4}
				}
			}
		}
	}
}
