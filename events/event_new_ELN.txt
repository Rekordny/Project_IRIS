country_event = {
	id = ELN_exile.3
	title = ELN_exile.3.t
	desc = ELN_exile.3.desc
	# picture = GFX_ELN_exile_3
	is_triggered_only = yes
	trigger = {
	}
	option = {
		name = ELN_exile.3.a
		ai_chance = {
			base = 1
		}
		every_country = {
			limit = {
				is_ally_with = ROOT
				original_tag = PEN
			}
			if = {
				#是自己的傀儡
				limit = {
					has_subject = ROOT
				}
				PREV = {
					transfer_navy = {
						target = FROM
					}
				}
				FROM = {
					annex_country = {
						target = PREV
						transfer_troops = yes
					}
				}
			}
			else_if = {
				#是小弟
				limit = {
					is_puppet = no
					is_ally_with = ROOT
					ROOT = {
						is_faction_leader = yes
					}
				}
				PREV = {
					transfer_navy = {
						target = FROM
					}
				}
				FROM = {
					annex_country = {
						target = PREV
						transfer_troops = yes
					}
				}
			}
		}
		every_state = {
			limit = {
				is_core_of = FROM
				is_owned_by = ROOT
			}
			transfer_state_to = FROM
		}
	}
	option = {
		name = ELN_exile.3.b
		ai_chance = {
			base = 0
		}
		FROM = {
			country_event = {
				id = ELN_exile.30001
			}
		}
	}
}
