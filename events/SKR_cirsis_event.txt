country_event = {#擦枪走火
    id = SKR_crisis.7
    title = SKR_crisis.7.t
    desc = SKR_crisis.7.desc
    #picture = GFX_event_SKR_crisis_7
    trigger = {
        tag = USF
        has_global_flag = SKR_CRISIS_FLAG
        NOT = {
            has_global_flag = SKR_CRISIS_ENDED_FLAG
        }
        NOT = {
            has_global_flag = ROSSIYA_SHIP_ATTACKED
        }
        has_country_flag = SKR_fire_at_will_flag

        check_variable = {ZTZ.SKR_tension > 70}
        OR = {
            AND = {
                country_exists = NAC
                NOT = {country_exists = BLA}
                NOT = {country_exists = OPA}
            }
            AND = {
                country_exists = BLA
                NOT = {country_exists = NAC}
                NOT = {country_exists = OPA}
            }
            AND = {
                country_exists = OPA
                NOT = {country_exists = BLA}
                NOT = {country_exists = NAC}
            }
        }
    }
    fire_only_once = yes

    mean_time_to_happen = {
        days = 300
        modifier = {
            factor = 0.2
            check_variable = {ZTZ.SKR_tension > 80}
        }
        modifier = {
            factor = 0.2
            check_variable = {ZTZ.SKR_tension > 90}
        }
    }

    immediate = {
        set_global_flag = ROSSIYA_SHIP_ATTACKED
    }

    option = {
        name = SKR_crisis.7.a

        custom_effect_tooltip = SKR_crisis_7_t
        hidden_effect = {
            news_event = {id = SKR_crisis_news.1 days = 2}
            SLL = {
                add_to_variable = {power_projection_m = 6}
            }
            if = {
                limit = {
                    country_exists = NAC
                    NOT = {country_exists = BLA}
                }
                NAC = {
                    set_temp_variable = { tag_temp = SKR }
		            IRIS_activate_world_war_member = yes
                }
            }
            else_if = {
                limit = {
                    country_exists = BLA
                    NOT = {country_exists = NAC}
                }
                BLA = {
                    set_temp_variable = { tag_temp = SKR }
		            IRIS_activate_world_war_member = yes
                }
            }
            else_if = {
                limit = {
                    country_exists = OPA
                    NOT = {country_exists = BLA}
                    NOT = {country_exists = NAC}
                }
                OPA = {
                    set_temp_variable = { tag_temp = SKR }
		            IRIS_activate_world_war_member = yes
                }
            }
        }

        ai_chance = {
            base = 10
        }
    }
}
