country_event = {
	id = YMI.12
	title = YMI.12.t
	desc = YMI.12.desc

	is_triggered_only = yes


	option = {
		name = YMI.12.a
		if = {
			limit = {
				country_exists = ATM
			}
			add_to_war = {
				targeted_alliance = ATM
				enemy = ACD
				hostility_reason = asked_to_join
			}
			give_military_access = ATM
			ATM = {
				give_military_access = ROOT
				country_event = {id = YMI.13}
			}
		}
		if = {
			limit = {
				country_exists = ATS
			}
			add_to_war = {
				targeted_alliance = ATS
				enemy = ACD
				hostility_reason = asked_to_join
			}
			give_military_access = ATS
			ATS = {
				give_military_access = ROOT
				country_event = {id = YMI.13}
			}
		}
		ai_will_do = {
			base = 10
		}
	}
	option = {
		name = YMI.12.b
		
		if = {
			limit = {
				country_exists = ATS
			}
			ATS = {
				add_manpower = 4000
				add_equipment_to_stockpile = {
					type = refined_mana
					amount = 50
					producer = YMI
				}
				add_equipment_to_stockpile = {
					type = mana_infantry_weapons
					amount = 500
					producer = YMI
				}
				add_equipment_to_stockpile = {
					type = Magic_ap_equipment
					amount = 250
					producer = YMI
				}
				country_event = {id = YMI.14}
			}
		}
		else_if = {
			limit = {
				country_exists = ATM
			}
			ATM = {
				add_manpower = 4000
				add_equipment_to_stockpile = {
					type = refined_mana
					amount = 50
					producer = YMI
				}
				add_equipment_to_stockpile = {
					type = mana_infantry_weapons
					amount = 500
					producer = YMI
				}
				add_equipment_to_stockpile = {
					type = Magic_ap_equipment
					amount = 250
					producer = YMI
				}
				country_event = {id = YMI.14}
			}
		}
		ai_will_do = {
			base = 2
		}
	}
	option = {
		name = YMI.12.c
		add_war_support = -0.05
		ai_will_do = {
			base = 1
		}
	}
}
